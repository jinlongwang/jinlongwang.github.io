  抽象类
  example：
      class Interface(object):
        def __init__(self):
            pass

        def toImpl(self):
            raise NotImplementedError("must has a sub class")

      class Sub(Interface):
        def toImpl(self):
            print 'impl ok!'

      if __name__ == "__main__":
        b = Sub()
        b.toImpl()
